### 離線安裝docker、docker-compose
collapsed:: true
	- 下載 [docker](https://download.docker.com/linux/static/stable/)
	  ```
	  # 解壓 tar 包
	  tar -xvf docker-29.1.3.tgz
	  
	  # 將 docker 二進制文件移到 /usr/bin/
	  sudo cp docker/* /usr/bin/
	  
	  # 創建 docker.service 文件
	  sudo tee /etc/systemd/system/docker.service > /dev/null << 'EOF'
	  [Unit]
	  Description=Docker Application Container Engine
	  Documentation=https://docs.docker.com
	  After=network-online.target firewalld.service
	  Wants=network-online.target
	  
	  [Service]
	  Type=notify
	  ExecStart=/usr/bin/dockerd
	  ExecReload=/bin/kill -s HUP $MAINPID
	  LimitNOFILE=infinity
	  LimitNPROC=infinity
	  LimitCORE=infinity
	  TimeoutStartSec=0
	  Delegate=yes
	  KillMode=process
	  Restart=on-failure
	  StartLimitBurst=3
	  StartLimitInterval=60s
	  
	  [Install]
	  WantedBy=multi-user.target
	  EOF
	  
	  # 添加文件權限
	  sudo chmod +x /etc/systemd/system/docker.service
	  
	  # 重新加載 systemd 配置
	  sudo systemctl daemon-reload
	  
	  # 啟動 Docker
	  sudo systemctl start docker
	  
	  # 設置開機自啟
	  sudo systemctl enable docker.service
	  
	  # 驗證安裝
	  docker -v
	  ```
	- 下載 [docker-compose](https://github.com/docker/compose/releases/)
	  ```
	  # 移動二進制文件到系統路徑
	  sudo mv docker-compose-linux-x86_64 /usr/local/bin/docker-compose
	  
	  # 賦予執行權限
	  sudo chmod +x /usr/local/bin/docker-compose
	  
	  # 驗證安裝
	  docker-compose --version
	  ```
